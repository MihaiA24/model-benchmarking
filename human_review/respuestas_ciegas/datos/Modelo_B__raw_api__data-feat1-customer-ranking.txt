```python
"""
Tarea: rankear clientes por total comprado dentro de su país usando window functions.
Columnas requeridas: Country, CustomerId, FirstName, LastName, TotalPurchases, Rank
- TotalPurchases = suma de Invoice.Total por cliente, redondeada a 2 decimales

"""
import sqlite3
import pandas as pd

conn = sqlite3.connect('Chinook.db')

# Query con window function RANK(): se calcula primero la suma por cliente
# y luego se aplica RANK() particionado por país, ordenado por TotalPurchases DESC
df = pd.read_sql_query("""
    WITH CustomerTotals AS (
        SELECT
            c.Country,
            c.CustomerId,
            c.FirstName,
            c.LastName,
            ROUND(SUM(i.Total), 2) AS TotalPurchases
        FROM Customer c
        JOIN Invoice i ON c.CustomerId = i.CustomerId
        GROUP BY c.Country, c.CustomerId, c.FirstName, c.LastName
    )
    SELECT
        Country,
        CustomerId,
        FirstName,
        LastName,
        TotalPurchases,
        RANK() OVER (PARTITION BY Country ORDER BY TotalPurchases DESC) AS Rank
    FROM CustomerTotals
    ORDER BY Country ASC, Rank ASC
""", conn)

conn.close()
print(df.to_string(index=False))
```
